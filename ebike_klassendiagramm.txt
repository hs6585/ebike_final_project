abstract class BatteryBase {
  + C_nom : float
  + soc : float
  + R_int : float
  + Vmin : float
  + Vmax : float
  --
  + {abstract} __init__(capacity_nom_Ah: float, initial_soc: float)
  + {abstract} apply_current(current: float, duration: float) : None
  + {abstract} voltage(current: float) : float
}

class BatteryPack {
  + C_nom_Ah : float
  + C_nom : float
  + soc : float
  + R_int : float
  + Vmin : float
  + Vmax : float
  --
  + __init__(capacity_nom_Ah, internal_resistance_mOhm, initial_soc, Vmin, Vmax, cells_in_parallel)
  + apply_current(current: float, duration: float) : None
  + voltage(current: float) : float
  + is_empty() : bool
  + is_full() : bool
  + __str__() : str
}

class LiPoBatteryPack {
  + list_soc : list[float]
  + list_uoc : list[float]
  --
  + __init__(capacity_nom_Ah, internal_resistance_mOhm, initial_soc, Vmin, Vmax, cells_in_parallel)
  + voltage(current: float) : float
}

class NMCBatteryPack {
  + list_soc : list[float]
  + list_uoc : list[float]
  --
  + __init__(capacity_nom_Ah, internal_resistance_mOhm, initial_soc, Vmin, Vmax, cells_in_parallel)
  + voltage(current: float) : float
}

BatteryBase <|-- BatteryPack
BatteryPack <|-- LiPoBatteryPack
BatteryPack <|-- NMCBatteryPack
